
  CREATE TABLE FLIGHT (
	fid INTEGER IDENTITY,
	flight_from VARCHAR(32) NOT NULL,
	flight_to VARCHAR(32) NOT NULL,
	date VARCHAR(20) NOT NULL
	
	
);

	CREATE TABLE CUSTOMER (
	id INTEGER IDENTITY, 
	flightid INTEGER ,
	fname VARCHAR(32) NOT NULL,
	lname VARCHAR(32) NOT NULL,
	passportno VARCHAR(20) NOT NULL,
	gender VARCHAR(10) NOT NULL,
	seatno VARCHAR(5) NOT NULL,
	cabintype VARCHAR(15) NOT NULL,
	price INTEGER ,
	
	FOREIGN KEY (flightid) REFERENCES FLIGHT(fID)ON DELETE CASCADE ON UPDATE CASCADE
	
);
